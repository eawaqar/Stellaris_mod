country_event = {
    id = easybuff.1
    title = "Easybuff Menu"
	desc = "easybuff.1.desc"
	picture = GFX_evt_society_research
    is_triggered_only = yes
    trigger = { is_ai = no }

    option = {
        name = ToggleDefaultBuff

            limit = { has_modifier = "modifier_easybuf_default" }

                add_modifier = {

    }

    option = {
        name = CountryBuffs
        hidden_effect = { country_event = { id = easybuff.14 } }
    }

    option = {
        name = RemoveAllBuffs

        name = CorvetteMIDCore

    }
    option = {
        name = PlanetKillerGaiaGenerator

    }
