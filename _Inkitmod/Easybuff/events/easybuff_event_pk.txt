namespace = planet_destruction

planet_event = {
	id = planet_destruction.700
	hide_window = yes

	is_triggered_only = yes

	immediate = {
		if = {
			limit = { is_star = no }
			set_planet_size = 25
			if = {
				limit = { NOT = { is_planet_class = pc_gaia } }
				change_pc = { class = pc_gaia }
				generate_easybuff_tiles = yes
				else = {
					generate_easybuff_tiles2 = yes
				}
			}
			else = {
				generate_easybuff_starres = yes
			}
		}
    }
}
